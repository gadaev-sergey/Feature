# gulp
Моя сборка gulp
## Структура папок и файлов

```
├── gulp                          # папка с настройками gulp
│   ├── config                    # Папка с конфигарациями для gulp
│   │   └── ftp.js                # Файл с параметрами ftp сервера
│   │   └── path.js               # Файл с прописанными путями к файлам
│   │   └── plugins.js            # Файл с плагинами для gulp
│   ├── tasks                     # Папка с заданиями для gulp
├── src                           # Исходники
│   ├── files                     # Папка для хранения иных ассетов - php, видео-файлы, favicon и т.д.
│   │   └── favicon.png           # Фавикон сайта
│   ├── fonts                     # Папка для хранения шрифтов в любых форматах
│   ├── html                      # Папка для хранения html-частей страницы
│   │   └── footer.html           # Подвал сайта
│   │   └── header.html           # Шапка сайта
│   ├── img                       # Папка для хранения картинок
│   │   └── logo.png              # Логотип
│   ├── js                        # Скрипты
│   │   ├── components            # js-компоненты
│   │   ├── vendor                # Папка для загрузки локальных версий библиотек
│   │   └── _components.js_       # Файл с подключениями компонентов
│   │   └── _functions.js_        # Файл с готовыми функциями на js
│   │   └── _vars.js_             # Файл с переменными проекта
│   │   └── _vendor.js            # Файл с подключениями библиотек
│   │   └── app.js                # Главный скрипт
│   ├── scss                      # Стили сайта (препроцессор sass в scss-синтаксисе)
│   │   ├── components            # scss-компоненты
│   │   ├── mixins                # Папка для сохранения готовых scss-компонентов
│   │   ├── vendor                # Папка для хранения локальных css-стилей библиотек
│   │   └── _mixins.scss          # Файл для подключения миксинов из папки mixins
│   │   └── _vars.scss            # Файл для написания scss-переменных
│   │   └── components.scss       # Файл для подключение scss-компонентов
│   │   └── fonts.scss            # Файл для подключения шрифтов
│   │   └── main.scss             # Главный файл стилей
│   │   └── settings.scss         # Файл для написания глобальных стилей
│   │   └── vendor.cscc           # Файл для подключения стилей библиотек из папки vendor
│   ├── svgicons                  # Папка для преобразования svg в спрайт
│   └── index.html                # Главный html-файл
└── .editorconfig                 # файл с настройками форматирования кода
└── gulpfile.js                   # файл с настройками Gulp
└── package.json                  # файл с настройками сборки и установленными пакетами
└── README.md                     # документация сборки
```
